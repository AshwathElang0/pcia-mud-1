\begin{frame}{Problem Setup}
\begin{itemize}
    \item \textbf{Objective}: Extract biological insights from a temporal sequence of colorimetric images ($0^{th}$ min to $25^{th}$ min).
    \item \textbf{System}: Bacterial-dye-antibiotic arrays (3x9 grid matrix).
    \item \textbf{Goal}: Determine the Minimum Inhibitory Concentration (MIC) and understand reaction kinetics.
    \item \textbf{Indicator}: Biological indicator dye transitions from blue (high $a^*$) to pink (higher $a^*$) as surviving bacteria metabolize it.
\end{itemize}
\end{frame}

\begin{frame}{Grid Detection \& Extraction}
\begin{itemize}
    \item \textbf{Histogram Projection}: Isolate bright sample wells from the dark tray background using grayscale conversion, thresholding, and morphology opening.
    \item \textbf{Peak Finding}: Identify well centers via prominent peaks in projected 1D histograms over X and Y axes.
    \item \textbf{Median Extraction}: Draw a 32-pixel radius circular mask at each identified well location and extract median intensity across RGB, CIELAB ($L^*a^*b^*$), and HSV color spaces.
\end{itemize}
\end{frame}

\begin{frame}{1. Visual Difference Analysis}
\begin{itemize}
    \item \textbf{Approach}: Compare first and last frame visually and via absolute difference map.
    \item \textbf{Reported result}: Mean absolute intensity difference was about 44.
    \item \textbf{Reported result}: About 4.2 million pixels changed by more than 20 intensity units.
    \item \textbf{Interpretation}: Visible change exists, and we need the pipelines to convert it into strong biological trend metrics.
\end{itemize}
\end{frame}

\begin{frame}{2. Circular ROI Median Color Extraction}
\begin{itemize}
    \item \textbf{Approach}: Detect grid peaks, place fixed-radius circular masks, extract median color per well.
    \item \textbf{Strength}: Robust to outliers compared with mean-based extraction.
    \item \textbf{Limitation}: Sensitive to misalignment when grid detection or geometry changes.
    \item \textbf{Result}: Stable replicate behavior (CV roughly 2\% to 9\%), but weak biological separation in $a^*$.
\end{itemize}
\end{frame}

\begin{frame}{3. Ratio-Normalized Metrics}
\begin{itemize}
    \item \textbf{Approach}: Use R/G and A/L ratios to reduce illumination sensitivity.
    \item \textbf{R/G results}: Mostly small negative shifts; strongest positive shift in one column (+0.0530).
    \item \textbf{A/L results}: All columns negative (about -0.023 to -0.107).
    \item \textbf{Interpretation}: Normalization confirms broad brightening effects and weak concentration-ordered biological contrast.
\end{itemize}
\end{frame}

\begin{frame}{4. Pinkness Percentile Analysis}
\begin{itemize}
    \item \textbf{Approach}: Track upper-tail pink pixels ($A_{\max}$, $A_{p90}$, $A_{p80}$) rather than medians.
    \item \textbf{Result}: $A_{\max}$ increased in all columns (+4.67 to +11.00).
    \item \textbf{Result}: $A_{p90}$ increased in all columns (+2.00 to +7.33).
    \item \textbf{Observation}: Median $a^*$ still showed weak/mixed behavior.
    \item \textbf{Interpretation}: Changes appear in localized high-intensity regions, not consistently across full well areas.
\end{itemize}
\end{frame}

\begin{frame}{5. Temporal MIC Identification}
\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
    \item \textbf{Method}: Automated kinetic slope detection on the $a^*$ (green-red) channel over time.
    \item \textbf{Threshold}: Monotonic linear regression slope ($m > 0.1$) indicates bacterial survival.
    \item \textbf{Result}: Column 5 identified as the MIC (last completely inhibited concentration).
\end{itemize}

\begin{frame}{8. Early Prediction \& Curvilinear Kinetics}
\begin{itemize}
    \item \textbf{Early Forecasting}: Logistic regression classifier achieved 85.7\% accuracy predicting final MIC outcome at $t=25$ using \textit{only} the initial $t=0$ image.
    \item \textbf{Curvilinear Kinetics}: Extracting $R^2$ exponential decay/growth coefficients using non-linear curve fits to overlying $a^*$ channel datapoints. Provides insight into deceleration rate and specific bacteriostatic/bactericidal mechanism.
\end{itemize}

\begin{frame}{10. Intra-Disk Region Detection}
\begin{columns}
\begin{column}{0.4\textwidth}
\begin{itemize}
    \item \textbf{Method}: K-Means clustering ($K=2$) spatially \textit{within} pixels of sample wells to isolate 'Reacted' (pink) from 'Unreacted' (blue).
    \item \textbf{Inference}: Growth isn't a uniform color fade. Pink region area percentage expands non-linearly, showing localized metabolic activity.

\begin{frame}{Conclusion}
\begin{itemize}
    \item Successfully automated MIC identification using robust, multi-channel colorimetric analysis.
    \item Advanced ML (PCA, K-Means) revealed deeper color channel insights \& non-linear behavioral responses.
    \item Spatial segmentation within discs proves growth is localized before diffusing.
    \item Early prediction models suggest feasibility of significantly reducing required incubation time.
\end{itemize}
\end{frame}

\begin{frame}{Consolidated Results and Practical Takeaways}
\begin{itemize}
    \item \textbf{Main result}: Strong visual change but weak statistically significant $a^*$ trend.
    \item \textbf{Most likely bottlenecks}: Image acquisition inconsistency, ROI alignment uncertainty, and illumination artifacts (large $L^*$ drift).
    \item \textbf{Recommendation 1}: Standardize imaging geometry and camera settings across all timepoints.
    \item \textbf{Recommendation 2}: Validate ROI placement on every frame or switch to segmentation-first extraction.
    \item \textbf{Recommendation 3}: Keep percentile and ratio metrics as secondary diagnostics.
    \item \textbf{Recommendation 4}: Extend incubation time and/or strengthen indicator chemistry for larger biological effect size.
\end{itemize}
\end{frame}
